SELECT  
HP.PARTY_NAME,
HP.PARTY_NUMBER,
HPS.PARTY_SITE_NAME,
PPNF.FULL_NAME AS Buyer,
PHA.Segment1,
PHA.APPROVED_DATE,
PLA.PRC_BU_ID,
PLA.REQ_BU_ID ,
PLA.ITEM_ID,
PLA.LINE_NUM ,
PLA.QUANTITY ,
PLA.LIST_PRICE_PER_UNIT,
PLA.UNIT_PRICE,
PLLA.NEED_BY_DATE ,
PLLA.PROMISED_DATE ,
PLLA.REQUESTED_SHIP_DATE,
PLLA.PROMISED_SHIP_DATE,
PLLA.CLOSED_FOR_RECEIVING_DATE ,
PLLA.QUANTITY_RECEIVED,
PLLA.QUANTITY_ACCEPTED,
PLLA.QUANTITY_REJECTED,
PDA.QUANTITY_CANCELLED ,
PDA.DESTINATION_SUBINVENTORY ,
PDA.QUANTITY_BILLED ,
PDA.AMOUNT_BILLED,
RSH.RECEIPT_NUM,
ESIB.ITEM_NUMBER,
PS.SEGMENT1 AS SEGMENT_1,
HOU.NAME AS PURCHASING_BU,
HOU.NAME AS REQUESTING_BU 
FROM HZ_PARTIES HP
JOIN HZ_PARTY_SITES HPS ON HPS.OBJECT_VERSION_NUMBER=HP.OBJECT_VERSION_NUMBER
JOIN PER_PERSON_NAMES_F PPNF ON PPNF.OBJECT_VERSION_NUMBER=HPS.OBJECT_VERSION_NUMBER
JOIN PO_HEADERS_ALL PHA ON PHA.OBJECT_VERSION_NUMBER=HPS.OBJECT_VERSION_NUMBER
JOIN PO_LINES_ALL PLA ON PLA.OBJECT_VERSION_NUMBER=HPS.OBJECT_VERSION_NUMBER
JOIN PO_LINE_LOCATIONS_ALL PLLA ON PLLA.REQ_BU_ID=PLA.REQ_BU_ID
JOIN PO_DISTRIBUTIONS_ALL PDA ON PDA.LINE_LOCATION_ID=PLLA.LINE_LOCATION_ID
JOIN RCV_SHIPMENT_HEADERS RSH ON RSH.OBJECT_VERSION_NUMBER=PDA.OBJECT_VERSION_NUMBER
JOIN EGP_SYSTEM_ITEMS_B ESIB ON ESIB.OBJECT_VERSION_NUMBER=RSH.OBJECT_VERSION_NUMBER
JOIN POZ_SUPPLIERS PS ON PS.PARTY_ID=HP.PARTY_ID
JOIN HR_ORGANIZATION_UNITS HOU ON HOU.OBJECT_VERSION_NUMBER=RSH.OBJECT_VERSION_NUMBER
WHERE
ROWNUM<100
